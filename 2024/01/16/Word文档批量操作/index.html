winform.btnOpen.oncommand = function(id,event){
	var path = fsys.dlg.dir(, winform, '请选择需操作的文件目录') if (not path) {
		return;
	}

	global.WpsFiles = {};
	fsys.enum(path, "*.*",
		function(dir, filename, fullpath, findData) {

			if (filename) {
				table.push(WpsFiles, fullpath);
			}
		},
		winform.chkTest.checked);
	winform.edit.text=path+" 下共有"+#WpsFiles+"个文件。";
	winform.edit2.text="";

// 更名
winform.btnRename.oncommand = function(id, event) {
	var ff,OldName,NewName;
	//var opendir=winform.edit1.text;
	winform.edit2.setsel(0,-1);
	winform.edit2.clear();

	for (i = 1;#WpsFiles; 1) {
		var path = WpsFiles[i];
		//var ext = string.match(path,"[^.]+$");
		//var ext = fsys.getExtensionName(path);
		var newPath = path + ".docx";

		if (io.exist(newPath)) {
			continue;
		}

		try {

            OldName=path
            NewName=string.replace(OldName,winform.edit3.text,winform.edit4.text)
            fsys.rename(OldName,NewName)
            var str="原文件名："+OldName+'\r\n'+"新文件名："+NewName+'\r\n'
	
            winform.edit2.appendText(str);
            winform.edit2.vScroll(7/*_SB_BOTTOM*/)
			//WpsFiles
		} catch(e) {
			//continue;
		}
		winform.progressplus.progressPos=(i / #WpsFiles) * 100;

	}

	if (WpsObject) {
		com.Release(WpsObject);
		WpsObject = nil;
	}

	winform.progressplus.text = "操作已完成！";

}
winform.show();
win.loopMessage();
return winform;

winform.btnFormatConv.oncommand = function(id, event) {
	var WpsObject = com.CreateObject("KWPS.Application");
	if (not WpsObject) {
		winform.msgbox("请先安装WPS专业版");
		return;
	}

	for (i = 1;#WpsFiles; 1) {
		var path = WpsFiles[i];
		//var ext = string.match(path,"[^.]+$");
		//var ext = fsys.getExtensionName(path);
		var newPath = path + ".docx";

		if (io.exist(newPath)) {
			continue;
		}

		try {
			WpsObject.Visible = false;
			WpsObject.Documents.Open(path);
			if(winform.radioDocx.checked){
				var docxname = string.replace(path,"\.doc$",".docx");
				WpsObject.ActiveDocument.SaveAs2(docxname, 16);
				WpsObject.ActiveDocument.Close();
				var str="原文件名："+path +'\r\n'+"新文件名："+docxname+'\r\n'
            	winform.edit2.appendText(str);
            	winform.edit2.vScroll(7/*_SB_BOTTOM*/)
            }
            elseif(winform.radioPdf.checked){
				var pdfname = string.replace(path,"\.doc$",".pdf");
				WpsObject.ActiveDocument.SaveAs2(pdfname, 17);
				WpsObject.ActiveDocument.Close();
				var str="原文件名："+path +'\r\n'+"新文件名："+pdfname+'\r\n'
            	winform.edit2.appendText(str);
            	winform.edit2.vScroll(7/*_SB_BOTTOM*/)            
            }
            elseif(winform.radioDoc.checked){
				var docname = string.replace(path,"\.docx",".doc");
				WpsObject.ActiveDocument.SaveAs2(docname, 0);
				WpsObject.ActiveDocument.Close();
				var str="原文件名："+path +'\r\n'+"新文件名："+docname+'\r\n'
            	winform.edit2.appendText(str);
            	winform.edit2.vScroll(7/*_SB_BOTTOM*/)            
            }
			//WpsFiles
		} catch(e) {
			//continue;
		}
		winform.progressplus.progressPos=(i / #WpsFiles) * 100;
		//sleep(100);
	}

	if (WpsObject) {
		com.Release(WpsObject);
		WpsObject = nil;
	}

	winform.progressplus.text = "操作已完成！";

}


winform.show();
win.loopMessage();
return winform;

winform.btnSaveSet.oncommand = function(id, event) {

	var WpsObject = com.CreateObject("KWPS.Application");
	if (not WpsObject) {
		winform.msgbox("请先安装WPS专业版");
		return;
	}

	for (i = 1;#WpsFiles; 1) {
		var path = WpsFiles[i];
		//var ext = string.match(path,"[^.]+$");
		//var ext = fsys.getExtensionName(path);
		var newPath = path + ".docx";

		if (io.exist(newPath)) {
			continue;
		}

		try {
			WpsObject.Visible = false;
			WpsObject.Documents.Open(path);
			if(winform.radioDocx.checked){
				var docxname = string.replace(path,"\.doc$",".docx");
				WpsObject.ActiveDocument.SaveAs2(docxname, 16);
				WpsObject.ActiveDocument.Close();
				var str="原文件名："+path +'\r\n'+"新文件名："+docxname+'\r\n'
            	winform.edit2.appendText(str);
            	winform.edit2.vScroll(7/*_SB_BOTTOM*/)
            }
            elseif(winform.radioPdf.checked){
				var pdfname = string.replace(path,"\.doc$",".pdf");
				WpsObject.ActiveDocument.SaveAs2(pdfname, 17);
				WpsObject.ActiveDocument.Close();
				var str="原文件名："+path +'\r\n'+"新文件名："+pdfname+'\r\n'
            	winform.edit2.appendText(str);
            	winform.edit2.vScroll(7/*_SB_BOTTOM*/)            
            }
            elseif(winform.radioDoc.checked){
				var docname = string.replace(path,"\.docx",".doc");
				WpsObject.ActiveDocument.SaveAs2(docname, 0);
				WpsObject.ActiveDocument.Close();
				var str="原文件名："+path +'\r\n'+"新文件名："+docname+'\r\n'
            	winform.edit2.appendText(str);
            	winform.edit2.vScroll(7/*_SB_BOTTOM*/)            
            }
			//WpsFiles
		} catch(e) {
			//continue;
		}
		winform.progressplus.progressPos=(i / #WpsFiles) * 100;
		//sleep(100);
	}

	if (WpsObject) {
		com.Release(WpsObject);
		WpsObject = nil;
	}

	winform.progressplus.text = "操作已完成！";

}
